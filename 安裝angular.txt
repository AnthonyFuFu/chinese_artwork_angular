// 安裝angular-cli套件
npm install -g @angular/cli

// 解除安裝
npm uninstall -g @angular/cli

// Powershell 執行策略
Set-ExecutionPolicy -Scope CurrentUser -ExecutionPolicy RemoteSigned

// 建立angular專案 命令提示字元
npx @angular/cli@latest new seal_carving

// 安裝專案依賴套件
npm install

// 啟動專案
npx ng serve --open

// 測試網址
http://localhost:4200/
